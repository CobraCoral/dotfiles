1. Updating GRUB
# info -f grub -n 'Simple configuration'
# sudo vim /etc/default/grub
# sudo grub-mkconfig
# sudo update-grub
# reboot

2. Updating X:
# sudo vim /etc/X11/xorg.conf
# startx

3. Getting Display Information:
```
echo "======== Dimensions"
xrandr
xdpyinfo  | grep -oP 'dimensions:\s+\K\S+'

echo "======== Video"
sudo lshw -c video
inxi -Fxz -G --display

echo "======== Vendor:"
glxinfo | grep vendor

echo "\n======== lspci VGA:"
lspci -k | grep -A 2 -i "VGA"

echo "\n"
sudo lspci -nnkvv | less
```
